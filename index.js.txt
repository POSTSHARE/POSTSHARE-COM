        <input type="text" id="postText" placeholder="What's on your mind?">
        <button onclick="addPost()">Post</button>
        <div id="feed"></div>
    </div>

    <script>
        function addPost() {
            let postText = document.getElementById("postText").value;
            if (postText === "") return;
            let post = document.createElement("div");
            post.className = "post";
            post.innerHTML = `<p>${postText}</p><button onclick="likePost(this)">Like (0)</button>`;
            document.getElementById("feed").prepend(post);
            document.getElementById("postText").value = "";
        }

        function likePost(button) {
            let count = parseInt(button.innerText.match(/\d+/)[0]) + 1;
            button.innerText = `Like (${count})`;
        }
    </script>
